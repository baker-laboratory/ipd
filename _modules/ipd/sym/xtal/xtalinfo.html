def xtalinfo(name):
    if _xtal_info_dict is None:
        _populate__xtal_info_dict()

    name = name.upper().strip()

    if name not in _xtal_info_dict:
        alternate_names = {
            "P432": "P 4 3 2",
            "P432_43": "P 4 3 2 43",
            "F432": "F 4 3 2",
            "I432": "I 4 3 2 432",
            "I432_432": "I 4 3 2 432",
            "I 4 3 2": "I 4 3 2 432",
            "I4132": "I 41 3 2",
            "P4132": "P 41 3 2",
            "P213": "P 21 3",
            "P213_33": "P 21 3",
            "I213_32": "I 21 3",
            "I213": "I 21 3",
            "L6M322": "L6M_322",
            "L632": "L6_32",
            "P6_32": "L6_32",
            "P4_42": "L4_42",
            "P4_44": "L4_44",
            "P3_33": "L3_33",
            "P23": "P 2 3",
        }

        if name in alternate_names:
            name = alternate_names[name]
        else:
            raise ValueError(f'unknown xtal {name}')
    if name not in _xtal_info_dict:  # type: ignore
        name = name.replace("_", " ")
    # ipd.icv(name)
    info = _xtal_info_dict[name]  # type: ignore
    return name, info
